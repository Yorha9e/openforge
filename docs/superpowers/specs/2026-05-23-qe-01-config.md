## 完整结构体 (12 字段，一次配完)

```go
// internal/agent/domain/query_engine_config.go

type QueryEngineConfig struct {
    // ── 基础 ──
    MaxToolRounds      int   // 最大工具调用轮数，默认 10 (防止无限 tool loop)
    TokenBudget        int   // 单次 Submit Token 预算，默认 200000
    CheckpointInterval int   // 自动检查点间隔轮数，默认 10 (0 = 禁用)

    // ── Gate 审批 ──
    GateTimeout        time.Duration // 审批超时，默认 5min，超时自动拒绝
    ConflictResolution string        // 审批期间 ArtifactHash 变化时的策略: "abort"(默认) | "override"

    // ── 工具错误策略 (引用 §3.15) ──
    ToolErrorPolicy    ToolErrorPolicy

    // ── 历史压缩 ──
    HistoryCompression HistoryCompressionConfig

    // ── 可观测性 ──
    EnableMetrics      bool          // 启用 Prometheus 指标，默认 false (Phase 7 开启)
    EnableTracing      bool          // 启用分布式追踪，默认 false (Phase 8 开启)

    // ── P2 自动审批 (Phase 6+) ──
    AutoApprovalRules  []AutoApprovalRule // 低风险操作自动跳过 Gate
}

// DefaultQueryEngineConfig — Phase 1.5/3 生产默认值
var DefaultQueryEngineConfig = QueryEngineConfig{
    MaxToolRounds:      10,
    TokenBudget:        200000,
    CheckpointInterval: 10,
    GateTimeout:        5 * time.Minute,
    ConflictResolution: "abort",
    ToolErrorPolicy:    DefaultToolErrorPolicy,
    HistoryCompression: HistoryCompressionConfig{
        Enabled:          false,     // Phase 1.5 禁用 (Phase 3 开启)
        TriggerThreshold: 0.8,
        KeepRecentRounds: 10,
        MaxSummaryTokens: 2000,
    },
    EnableMetrics:      true,
    EnableTracing:      false,
    AutoApprovalRules:  nil,         // Phase 6+ 加载
}
```

## 子结构体定义

```go
// ToolErrorPolicy — 工具执行失败时的重试与降级策略 (统一 §3.15)
type ToolErrorPolicy struct {
    MaxRetries    int           // 最大重试次数，默认 3
    RetryDelay    time.Duration // 初始重试间隔，默认 1s
    BackoffFactor float64       // 指数退避因子，默认 2.0
    // OnFailure: "notify_llm"(默认) → 失败作为 tool_result 反馈 LLM
    //            "skip"           → 跳过继续
    //            "abort"          → 升到 §3.15 恢复链
    OnFailure     string
}

var DefaultToolErrorPolicy = ToolErrorPolicy{
    MaxRetries:    3,
    RetryDelay:    1 * time.Second,
    BackoffFactor: 2.0,
    OnFailure:     "notify_llm",
}

// HistoryCompressionConfig — 长对话历史压缩策略
type HistoryCompressionConfig struct {
    Enabled           bool    // 是否启用，默认 false
    TriggerThreshold  float64 // 触发阈值 (token 使用比例)，默认 0.8
    KeepRecentRounds  int     // 保留最近轮数，默认 10
    MaxSummaryTokens  int     // 摘要最大 token，默认 2000
}

// AutoApprovalRule — 自动审批规则 (P2, Phase 6+)
type AutoApprovalRule struct {
    Name        string
    Description string
    Condition   func(tool ToolCallParsed) bool
}
```
