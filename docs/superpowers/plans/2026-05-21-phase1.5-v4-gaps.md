type CSPChannel struct {

	walTail *walEntry

}

- [ ] **Step 2: 修改 Send — 通道满时写 WAL 而非丢弃**

```go
func (c *CSPChannel) Send(ctx context.Context, msg Message) error {
	select {
	case c.ch <- msg:
		return nil
	default:
	}

	c.mu.Lock()
	if c.walLen >= maxWALSize {
		c.mu.Unlock()
		return fmt.Errorf("channel %s full: WAL overflow (%d entries)", c.name, c.walLen)
	}
	entry := &walEntry{msg: msg}
	if c.walTail != nil {
		c.walTail.next = entry
	} else {
		c.walHead = entry
	}
	c.walTail = entry
	c.walLen++
	c.mu.Unlock()
	return nil
}
```

- [ ] **Step 3: 新增 Drain 方法 — 消费 WAL 到 channel**

```go
func (c *CSPChannel) Drain(ctx context.Context) {
	c.mu.Lock()
	entry := c.walHead
	c.walHead = nil
	c.walTail = nil
	c.walLen = 0
	c.mu.Unlock()

	for entry != nil {
		select {
		case c.ch <- entry.msg:
		case <-ctx.Done():
			return
		}
		entry = entry.next
	}
}
```

		t.Fatalf("WAL len = %d, want 4", ch.WALLen())

	// 消费 channel 中那条 → 腾出 slot
	<-ch.Receive()

	// Drain → WAL 消息入 channel
	ch.Drain(context.Background())

	// 现在应该能收到 4 条
	count := 0

		case <-ch.Receive():

		default:

		}
	}

	}
}

type goroutineEventBus struct {
	mu   sync.RWMutex
	subs map[string][]chan kernel.Event
}

func (b *goroutineEventBus) Publish(_ context.Context, topic string, event kernel.Event) error {
	b.mu.RLock()
	defer b.mu.RUnlock()
	for _, ch := range b.subs[topic] {
		select {
		case ch <- event:
		default:
		}
	}
	return nil
}

func (b *goroutineEventBus) Subscribe(_ context.Context, topic string) (<-chan kernel.Event, error) {
	b.mu.Lock()
	defer b.mu.Unlock()
	ch := make(chan kernel.Event, 64)
	b.subs[topic] = append(b.subs[topic], ch)
	return ch, nil
}
```

	c.mu.RLock()
	defer c.mu.RUnlock()
	v, ok := c.data[key]
	if !ok {
		return nil, fmt.Errorf("key %q not found", key)
	}
	return v, nil
}

func (c *memoryCache) Set(_ context.Context, key string, val any, _ time.Duration) error {
	c.mu.Lock()
	defer c.mu.Unlock()
	c.data[key] = val
	return nil
}

func (c *memoryCache) Del(_ context.Context, key string) error {
	c.mu.Lock()
	defer c.mu.Unlock()
	delete(c.data, key)
	return nil
}

	r.mu.RLock()
	defer r.mu.RUnlock()
	addrs, ok := r.services[name]
	if !ok {
		return nil, fmt.Errorf("service %q not found", name)

	copied := make([]string, len(addrs))

	return copied, nil

// internal/agent/port/tool.go
package port

import "context"

// Tool is the generic interface all tools must implement.
type Tool[Input any, Output any] interface {
	Name()             string
	Description()      string
	InputSchema()      []byte
	IsConcurrencySafe() bool
	IsReadOnly()       bool
	Execute(ctx context.Context, input Input) (Output, error)
}

// StreamingTool adds streaming execution capability.
type StreamingTool[Input any, Output any] interface {
	Tool[Input, Output]
	ExecuteStream(ctx context.Context, input Input) (<-chan StreamChunk[Output], error)
}

type StreamChunk[T any] struct {
	Value T
	Err   error
	Done  bool
}

// ToolMatch is returned by embedding-based tool search.
type ToolMatch struct {
	Tool            Tool[any, any]
	SimilarityScore float64
}

// ToolRegistry manages tool registration and discovery.
type ToolRegistry interface {
	Register(tool Tool[any, any]) error
	Search(ctx context.Context, query string, topK int) ([]ToolMatch, error)
	Get(name string) (Tool[any, any], error)
	List() []Tool[any, any]
}
```

// internal/auth/domain/permission_mode.go
package domain

import "context"

// PermissionMode determines the level of automation for Agent operations.
type PermissionMode int

const (
	PermissionModeDefault PermissionMode = iota // all writes require Gate approval
	PermissionModePlan                           // read-only ops auto-allowed
	PermissionModeAuto                           // rule-engine auto (file lock × whitelist)
	PermissionModeBypass                         // admin override, fully audited
)

func (m PermissionMode) String() string {
	switch m {
	case PermissionModeDefault: return "default"
	case PermissionModePlan:    return "plan"
	case PermissionModeAuto:    return "auto"
	case PermissionModeBypass:  return "bypass"
	default:                    return "unknown"
	}
}

// PermissionDecision is the outcome of a permission check.
type PermissionDecision int

const (
	DecisionAllow PermissionDecision = iota
	DecisionDeny
	DecisionAskGate // trigger Gate approval
)

// PermissionContext carries the full state for a permission check.
type PermissionContext struct {
	Mode      PermissionMode
	ToolName  string
	IsReadOnly bool
	FileInLock bool // file lock registry holds this path
	FileInWhitelist bool
}

// Classify decides whether a tool call should be allowed, denied, or escalated to Gate.
func Classify(ctx PermissionContext) PermissionDecision {
	if ctx.Mode == PermissionModeBypass {
		return DecisionAllow
	}
	if ctx.Mode == PermissionModePlan && ctx.IsReadOnly {
		return DecisionAllow
	}
	if ctx.Mode == PermissionModeAuto {
		if ctx.IsReadOnly || ctx.FileInLock || ctx.FileInWhitelist {
			return DecisionAllow
		}
		return DecisionAskGate
	}
	// default: everything requires Gate
	return DecisionAskGate
}

// SelectMode picks the initial PermissionMode for a Pipeline stage.
func SelectMode(pipelineLevel string, stage string) PermissionMode {
	if stage == "clarify" {
		return PermissionModePlan // read-only analysis
	}
	if pipelineLevel == "L1" || pipelineLevel == "L2" {
		return PermissionModeAuto // small changes, rule-engine
	}
	return PermissionModeDefault // L3/L4: full Gate
}
```

// internal/agent/domain/error_recovery.go
package domain

import (
	"context"
	"fmt"
)

// FailureCode mirrors the 3.14 failure classification.
type FailureCode string

const (
	FailModelHallucination FailureCode = "MODEL_HALLUCINATION"
	FailPromptWeakness     FailureCode = "PROMPT_WEAKNESS"
	FailDependencyConflict FailureCode = "DEPENDENCY_CONFLICT"
	FailSandboxTimeout     FailureCode = "SANDBOX_TIMEOUT"
	FailRepoBug            FailureCode = "REPO_BUG"
	FailContextOverflow    FailureCode = "CONTEXT_OVERFLOW"
	FailTokenQuotaExceeded FailureCode = "TOKEN_QUOTA_EXCEEDED"
	FailAPITimeout         FailureCode = "API_TIMEOUT"
	FailRateLimited        FailureCode = "RATE_LIMITED"
	FailOverloaded         FailureCode = "OVERLOADED"
	FailUnknown            FailureCode = "UNKNOWN"
)

// RecoveryAction describes what the recovery chain should do.
type RecoveryAction int

const (
	ActionRetry        RecoveryAction = iota // retry with exponential backoff
	ActionCompress                           // compress context and retry
	ActionDowngradeModel                     // switch to cheaper model
	ActionSelfRepair                         // Agent re-generates from repo baseline
	ActionClarify                            // Agent asks PM a clarifying question
	ActionLockVersion                        // Agent pins dependency version
	ActionEscalate                           // save context, pause pipeline, notify human
)

func (a RecoveryAction) String() string {
	switch a {
	case ActionRetry:         return "RETRY"
	case ActionCompress:      return "COMPRESS"
	case ActionDowngradeModel: return "DOWNGRADE_MODEL"
	case ActionSelfRepair:    return "SELF_REPAIR"
	case ActionClarify:       return "CLARIFY"
	case ActionLockVersion:   return "LOCK_VERSION"
	case ActionEscalate:      return "ESCALATE"
	default:                  return "UNKNOWN"
	}
}

// RecoveryResult is the outcome of a recovery attempt.
type RecoveryResult struct {
	Action  RecoveryAction
	Message string
}

// ClassifyAndRecover maps a failure to the appropriate recovery layer.
// Returns ActionEscalate if no automatic recovery applies.
func ClassifyAndRecover(code FailureCode, attempt int) RecoveryResult {
	if attempt < 0 {
		attempt = 0
	}

	// Layer 1: TRANSIENT — retry with backoff
	switch code {
	case FailAPITimeout, FailRateLimited, FailOverloaded:
		if attempt >= 3 {
			return RecoveryResult{ActionEscalate, fmt.Sprintf("TRANSIENT exhausted after %d retries", attempt)}
		}
		return RecoveryResult{ActionRetry, fmt.Sprintf("attempt %d/3", attempt+1)}
	}

	// Layer 2: DEGRADABLE
	if code == FailContextOverflow && attempt == 0 {
		return RecoveryResult{ActionCompress, "context compressed, retrying"}
	}
	if code == FailTokenQuotaExceeded && attempt == 0 {
		return RecoveryResult{ActionDowngradeModel, "downgrading to cheaper model"}
	}

	// Layer 3: RECOVERABLE
	switch code {
	case FailModelHallucination:
		return RecoveryResult{ActionSelfRepair, "re-generating from repo baseline"}
	case FailPromptWeakness:
		return RecoveryResult{ActionClarify, "asking PM for clarification"}
	case FailDependencyConflict:
		return RecoveryResult{ActionLockVersion, "locking dependency versions"}
	}

	// Layer 4: FATAL
	return RecoveryResult{ActionEscalate, fmt.Sprintf("FATAL: %s", code)}
}

// IsRetryable returns true if the failure can be retried at a lower layer.
func IsRetryable(code FailureCode) bool {
	switch code {
	case FailAPITimeout, FailRateLimited, FailOverloaded,
		FailContextOverflow, FailTokenQuotaExceeded,
		FailModelHallucination, FailPromptWeakness, FailDependencyConflict:
		return true
	default:
		return false
	}
}
```

	}
}

// StreamEvent is a single event emitted by the query loop.
type StreamEvent struct {
	Type    string // "delta" | "done" | "error"
	Content string
	Error   error
}

// QueryEngine manages the lifecycle of a single conversation.
type QueryEngine struct {
	mu               sync.Mutex
	state            QueryState
	llmClient        port.LLMRouterClient
	messageHistory   []port.Message
	tokenBudget      int
	tokenUsed        int
	config           port.LLMConfig
}

func NewQueryEngine(llmClient port.LLMRouterClient, config port.LLMConfig) *QueryEngine {
	return &QueryEngine{
		state:          QueryStateIdle,
		llmClient:      llmClient,
		tokenBudget:    config.MaxTokens,
		config:         config,
		messageHistory: make([]port.Message, 0),
	}
}

func (qe *QueryEngine) State() QueryState {
	qe.mu.Lock()
	defer qe.mu.Unlock()
	return qe.state
}

func (qe *QueryEngine) TokenUsed() int {
	qe.mu.Lock()
	defer qe.mu.Unlock()
	return qe.tokenUsed
}

// SubmitMessage sends a user message and returns a stream of response events.
func (qe *QueryEngine) SubmitMessage(ctx context.Context, msg string) (<-chan StreamEvent, error) {
	qe.mu.Lock()
	if qe.state != QueryStateIdle && qe.state != QueryStateAwaitingUser {
		qe.mu.Unlock()
		return nil, fmt.Errorf("cannot submit message in state %s", qe.state)
	}
	qe.state = QueryStateAwaitingLLM
	qe.messageHistory = append(qe.messageHistory, port.Message{Role: "user", Content: msg})
	qe.mu.Unlock()

	out := make(chan StreamEvent, 64)

	go func() {
		defer close(out)

		ch, err := qe.llmClient.ChatStream(ctx, port.ChatRequest{
			Messages: qe.messageHistory,
			Config:   qe.config,
		})
		if err != nil {
			qe.mu.Lock()
			qe.state = QueryStateError
			qe.mu.Unlock()
			out <- StreamEvent{Type: "error", Error: fmt.Errorf("llm stream: %w", err)}
			return
		}

		var fullText string
		for delta := range ch {
			fullText += delta
			out <- StreamEvent{Type: "delta", Content: delta}
		}

		qe.mu.Lock()
		qe.messageHistory = append(qe.messageHistory, port.Message{Role: "agent", Content: fullText})
		qe.tokenUsed += len(fullText) / 4 // rough estimate
		qe.state = QueryStateAwaitingUser
		qe.mu.Unlock()

		out <- StreamEvent{Type: "done", Content: fullText}
	}()

	return out, nil
}

// Messages returns a copy of the conversation history.
func (qe *QueryEngine) Messages() []port.Message {
	qe.mu.Lock()
	defer qe.mu.Unlock()
	cp := make([]port.Message, len(qe.messageHistory))
	copy(cp, qe.messageHistory)
	return cp
}

// Clear resets the conversation.
func (qe *QueryEngine) Clear() {
	qe.mu.Lock()
	defer qe.mu.Unlock()
	qe.state = QueryStateIdle
	qe.messageHistory = nil
	qe.tokenUsed = 0
}
